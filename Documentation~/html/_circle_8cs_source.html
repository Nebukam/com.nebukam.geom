82 public bool IntersectsXY(Circle circle)
83 {
84
85 float3 otherCenter = circle.center;
86
87 float dx = (otherCenter.x - center.x),
88 dy = (otherCenter.y - center.y),
89 d = sqrt(dx * dx + dy * dy),
90 otherRadius = circle.radius;

132 public bool TryGetIntersectionXY(Circle circle, out Segment line, float height = 0f)
133 {
134
135 float3 otherCenter = circle.center;
136
137 float dx = (otherCenter.x - center.x),
138 dy = (otherCenter.y - center.y),
139 d = sqrt(dx * dx + dy * dy),
140 otherRadius = circle.radius;
141
142 if (d <= (radius + otherRadius) && d >= abs(otherRadius - radius))
143 {
144
145 float rr = radius * radius,
146 ex = (otherCenter.x - center.x) / d,
147 ey = (otherCenter.y - center.y) / d,
148 x = (rr - otherRadius * otherRadius + d * d) / (2 * d),
149 y = sqrt(rr - x * x),
150 xex = center.x + x * ex, xey = center.y + x * ey,
151 yex = y * ex, yey = y * ey;
152
153 line = new Segment(
154 new float3(xex - yey, xey + yex, height),
155 new float3(xex + yey, xey - yex, height)
156 );
157
158 return true;
159
160 }
161 else
162 {
163 // No Intersection, far outside or one circle within the other
164 line = Segment.zero;
165 return false;
166 }
167 }

241 public int TryGetIntersectionXY(float3 A, float3 B,
242 out float3 intersection1, out float3 intersection2, float height = 0f)
243 {
244
245 float
246 cx = center.x,
247 cy = center.y,
248 Ax = A.x, Ay = A.y, Bx = B.x, By = B.y,
249 dx = Bx - Ax,
250 dy = By - Ay,
251 magA = dx * dx + dy * dy,
252 magB = 2f * (dx * (Ax - cx) + dy * (Ay - cy)),
253 magC = (Ax - cx) * (Ax - cx) + (Ay - cy) * (Ay - cy) - radius * radius,
254 det = magB * magB - 4f * magA * magC,
255 sqDet, t;
256
257 if ((magA <= float.Epsilon) || (det < 0f))
258 {
259 // No real solutions.
260 intersection1 = float3(float.NaN, float.NaN, height);
261 intersection2 = float3(float.NaN, float.NaN, height);
262
263 return 0;
264 }
265
266 if (det == 0)
267 {
268 // One solution.
269 t = -magB / (2f * magA);
270
271 intersection1 = float3(Ax + t * dx, Ay + t * dy, height);
272 intersection2 = float3(float.NaN, float.NaN, height);
273
274 return 1;
275 }
276 else
277 {
278 // Two solutions.
279 sqDet = sqrt(det);
280
281 t = ((-magB + sqDet) / (2f * magA));
282 intersection1 = float3(Ax + t * dx, Ay + t * dy, height);
283
284 t = ((-magB - sqDet) / (2f * magA));
285 intersection2 = float3(Ax + t * dx, Ay + t * dy, height);
286
287 return 2;
288 }
289 }
