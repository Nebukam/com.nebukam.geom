66 public bool ContainsXY(float3 pt)
67 {
68 float N = 1 / 2 * (-B.y * C.x + A.y * (-B.x + C.x) + A.x * (B.y - C.y) + B.x * C.y);
69 float sign = N < 0 ? -1 : 1;
70 float s = (A.y * C.x - A.x * C.y + (C.y - A.y) * pt.x + (A.x - C.x) * pt.y) * sign;
71 float t = (A.x * B.y - A.y * B.x + (A.y - B.y) * pt.x + (B.x - A.x) * pt.y) * sign;
72
73 return s > 0 && t > 0 && (s + t) < 2 * N * sign;
74 }

76 public bool ContainsXZ(float3 pt)
77 {
78 float N = 1 / 2 * (-B.z * C.x + A.z * (-B.x + C.x) + A.x * (B.z - C.z) + B.x * C.z);
79 float sign = N < 0 ? -1 : 1;
80 float s = (A.z * C.x - A.x * C.z + (C.z - A.z) * pt.x + (A.x - C.x) * pt.z) * sign;
81 float t = (A.x * B.z - A.z * B.x + (A.z - B.z) * pt.x + (B.x - A.x) * pt.z) * sign;
82
83 return s > 0 && t > 0 && (s + t) < 2 * N * sign;
84 }
