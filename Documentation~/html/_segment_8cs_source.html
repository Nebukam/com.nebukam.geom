115 public float3 ClosestPoint(float3 pt)
116 {
117
118 float3 C = float3(pt.x - A.x, pt.y - A.y, pt.z - A.z),//pt - A
119 D = float3(B.x - A.x, B.y - A.y, B.z - A.z);//B - A
120
121 float d = D.x * D.x + D.y * D.y + D.z * D.z; //Square distance
122
123 D = normalize(D);
124 float t = D.x * C.x + D.y * C.y + D.z * C.z; //Dot
125
126 if (t <= 0)
127 return A;
128
129 if ((t * t) >= d) //Distance check
130 return B;
131
132 C.x = A.x + D.x * t; C.y = A.y + D.y * t; C.z = A.z + D.z * t;
133 return C;

144 public bool IsIntersectingXY(Segment segment)
145 {
146 float2 A2 = float2(segment.A.x, segment.A.y), B2 = float2(segment.B.x, segment.B.y);
147
148 var d = (B.x - A.x) * (B2.y - A2.y) - (B.y - A.y) * (B2.x - A2.x);
149
150 if (d == 0.0f)
151 return false;
152
153 float u = ((A2.x - A.x) * (B2.y - A2.y) - (A2.y - A.y) * (B2.x - A2.x)) / d;
154 float v = ((A2.x - A.x) * (B.y - A.y) - (A2.y - A.y) * (B.x - A.x)) / d;
155
156 if (u < 0.0f || u > 1.0f || v < 0.0f || v > 1.0f)
157 return false;
158
159 return true;
160 }

275 public bool IsBetweenXY(float3 pt)
276 {
277
278 float2 ab = float2(B.x - A.x, B.y - A.y);//Entire line segment
279 float2 ac = float2(pt.x - A.x, pt.y - A.y);//The intersection and the first point
280
281 float dot = ab.x * ac.x + ab.y * ac.y;
282
283 //If the vectors are pointing in the same direction = dot product is positive
284 if (dot <= 0f) { return false; }
285
286 float abm = ab.x * ab.x + ab.y * ab.y;
287 float acm = ac.x * ac.x + ac.y * ac.y;
288
289 //If the length of the vector between the intersection and the first point is smaller than the entire line
290 return (abm >= acm);
291 }
