48 public UIntPair OppositeEdge(int v)

59 public bool SharesEdgeWith(Triad tri)
60 {
61 int oA = tri.A, oB = tri.B, oC = tri.C;
62 int i = 0;
63 if (A == oA || A == oB || A == oC) { i++; };
64 if (B == oA || B == oB || B == oC) { i++; };
65 if (C == oA || C == oB || C == oC) { i++; };
66 return (i == 2);
67 }
68
69 public bool SharesVertexWith(Triad tri)
70 {
71 return (A == tri.A || A == tri.B || A == tri.C
72 || B == tri.A || B == tri.B || B == tri.C
73 || C == tri.A || C == tri.B || C == tri.C);
74 }
75
76 public bool CircumcircleContainsXY(float3 v)
77 {
78 return ((v.x - circumcenter.x) * (v.x - circumcenter.x) + (v.y - circumcenter.y) * (v.y - circumcenter.y)) < sqRadius;
79 }
80
81 public bool CircumcircleContainsXZ(float3 v)
82 {
83 return ((v.x - circumcenter.x) * (v.x - circumcenter.x) + (v.z - circumcenter.z) * (v.z - circumcenter.z)) < sqRadius;
84 }
