284 private static void Triad(out Triad triad, int A, int B, int C)
285 {
286
287 float3 vA = vertices[A].pos, vB = vertices[B].pos, vC = vertices[C].pos, center;
288
289 float dA = vA.x * vA.x + vA.y * vA.y;
290 float dB = vB.x * vB.x + vB.y * vB.y;
291 float dC = vC.x * vC.x + vC.y * vC.y;
292
293 float aux1 = (dA * (vC.y - vB.y) + dB * (vA.y - vC.y) + dC * (vB.y -vA.y));
294 float aux2 = -(dA * (vC.x - vB.x) + dB * (vA.x - vC.x) + dC * (vB.x - vA.x));
295 float div = (2 * (vA.x * (vC.y - vB.y) + vB.x * (vA.y - vC.y) + vC.x * (vB.y - vA.y)));
296
297 if (div == 0)
298 {
299 //throw new System.Exception("div by zero.");
300 }
301
302 center = float3(aux1 / div, aux2 / div, 0f);
303 triad = new Triad(A, B, C, center, (center.x - vA.x) * (center.x - vA.x) + (center.y - vA.y) * (center.y - vA.y));
304
305 }
